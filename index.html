let peer, localStream, currentCall, timerInterval, vibeInterval; let dialString = ""; let isAudioUnlocked = false; // 音声ファイルの設定 const ringtone = new Audio("ring.mp3"); const dialtone = new Audio("dialtone.mp3"); // 発信音はループでOK、着信音は「通知音」風にするためループをOFFにする dialtone.loop = true; ringtone.loop = false; let myId = localStorage.getItem('nnm_phone_num') || "01-" + Math.floor(Math.random()*9000+1000); localStorage.setItem('nnm_phone_num', myId); window.onload = async () => { document.getElementById('my-number').innerText = myId; try { localStream = await navigator.mediaDevices.getUserMedia({ audio: true }); initPeer(); } catch(e) { alert("マイクを許可してください"); } }; function unlockAudio() { if (isAudioUnlocked) return; ringtone.play().then(() => { ringtone.pause(); ringtone.currentTime = 0; }); dialtone.play().then(() => { dialtone.pause(); dialtone.currentTime = 0; }); isAudioUnlocked = true; } function initPeer() { if (peer) peer.destroy(); peer = new Peer(myId, { debug: 1, config: { iceServers: [{ urls: "stun:stun.l.google.com:19302" }] } }); peer.on('open', (id) => { document.getElementById('my-number').innerText = id; document.getElementById('my-number').style.color = "var(--accent)"; }); peer.on('call', call => { currentCall = call; showCallOverlay(call.peer, "着信中...", true); // 着信音を「通知音」のように間隔をあけて鳴らす playRingtoneAsNotification(); if (navigator.vibrate) { vibeInterval = setInterval(() => navigator.vibrate([1000, 500]), 1500); } // 相手が発信をキャンセル（切断）した時の同期 call.on('close', () => { endCall(); }); }); peer.on('error', err => { if (err.type === 'unavailable-id') { myId = "01-" + Math.floor(Math.random()*9000+1000); localStorage.setItem('nnm_phone_num', myId); setTimeout(initPeer, 1000); } }); peer.on('disconnected', () => peer.reconnect()); } // 着信音を通知音っぽく鳴らす関数 let ringInterval; function playRingtoneAsNotification() { const play = () => { if (!currentCall) return; // 通話が終了してたら鳴らさない ringtone.play().catch(e => console.log("Audio locked")); }; play(); // 3秒おきに通知音を繰り返す（端末の通知音風） ringInterval = setInterval(play, 3000); } function makeCall(number, name = "発信中...") { if (!peer || peer.disconnected) { alert("ネットワークエラー"); return; } dialtone.play().catch(e => console.log("Unlock needed")); currentCall = peer.call(number, localStream); showCallOverlay(number, name, false); setupCallEvents(currentCall); } function answerCall() { stopSounds(); if (currentCall) { currentCall.answer(localStream); document.getElementById('btn-answer').style.display = 'none'; setupCallEvents(currentCall); } } function setupCallEvents(call) { call.on('stream', stream => { stopSounds(); document.getElementById('call-label').innerText = "通話中"; document.getElementById('timer').style.display = 'block'; const audio = document.createElement('audio'); audio.srcObject = stream; audio.autoplay = true; document.body.appendChild(audio); startTimer(); }); // 相手が切断したとき、自分も画面を閉じる（同期） call.on('close', () => { endCall(); }); call.on('error', () => { endCall(); }); } function stopSounds() { clearInterval(ringInterval); // 通知音リピート停止 ringtone.pause(); ringtone.currentTime = 0; dialtone.pause(); dialtone.currentTime = 0; clearInterval(vibeInterval); if (navigator.vibrate) navigator.vibrate(0); } // 拒否・終了ボタンを押したとき function endCall() { stopSounds(); if (currentCall) { currentCall.close(); // これにより相手側の call.on('close') が動く currentCall = null; } document.getElementById('call-overlay').style.display = 'none'; clearInterval(timerInterval); document.getElementById('timer').innerText = "00:00"; document.getElementById('timer').style.display = 'none'; } /* --- UI操作系（変更なし） --- */ function updateDialDisplay() { let display = dialString; if (dialString.length > 2) { display = dialString.substring(0, 2) + "-" + dialString.substring(2); } document.getElementById('dial-display').innerText = display; } function pressKey(k) { if (k === 'back') dialString = dialString.slice(0, -1); else if (dialString.length < 7) dialString += k; updateDialDisplay(); } function startDialCall() { const target = document.getElementById('dial-display').innerText; if (target.length < 4) return; makeCall(target); toggleKeypad(false); } function toggleKeypad(show) { document.getElementById('keypad-modal').classList.toggle('open', show); if(!show) { dialString = ""; updateDialDisplay(); } } function startTimer() { let sec = 0; clearInterval(timerInterval); timerInterval = setInterval(() => { sec++; let m = Math.floor(sec/60).toString().padStart(2,'0'); let s = (sec%60).toString().padStart(2,'0'); document.getElementById('timer').innerText = `${m}:${s}`; }, 1000); } function showCallOverlay(number, name, isIncoming) { document.getElementById('active-number').innerText = number; document.getElementById('active-name').innerText = name; document.getElementById('call-overlay').style.display = 'flex'; document.getElementById('btn-answer').style.display = isIncoming ? 'flex' : 'none'; document.getElementById('call-label').innerText = isIncoming ? "着信中..." : "発信中..."; }
